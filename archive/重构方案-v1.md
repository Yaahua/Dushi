# 重构方案：Narrative 管理系统 v1

## 目标

让代码易于理解、调试和维护。所有操作的行为显式声明，过期逻辑集中管理。

## 核心设计

### 1. 类型重定义

```typescript
// ============ types/narrative.ts ============

/** 叙事行类型 */
export type LineType = 'scene' | 'system' | 'player' | 'danger' | 'dialogue';

/** 交互链接 */
export interface ActionLink {
  label: string;
  command: string;
}

/** 叙事行 */
export interface NarrativeLine {
  id: string;
  type: LineType;
  text: string;
  links?: ActionLink[];
  expired?: boolean;
  metadata?: {
    actionType?: string;      // 'buy' | 'eat' | 'interact' | 'scene'
    locationId?: string;      // 关联的场景 ID
    timestamp: number;        // 创建时间
  };
}

/** 操作结果 */
export interface OperationResult {
  state: GameState;
  narrativeUpdates: NarrativeUpdate[];
  debug: OperationLog;
}

/** 叙事更新指令 */
export type NarrativeUpdate =
  | { type: 'add'; line: NarrativeLine }
  | { type: 'expireActionType'; actionType: string }
  | { type: 'expireLocation'; locationId: string }
  | { type: 'expireAllActions' }
  | { type: 'expireSceneLinks' };

/** 操作日志 */
export interface OperationLog {
  operation: string;
  timestamp: number;
  steps: string[];
}
```

### 3. 统一的叙事管理器

```typescript
// ============ managers/NarrativeManager.ts ============

export class NarrativeManager {
  private narrative: NarrativeLine[];
  private logs: string[];

  constructor(narrative: NarrativeLine[]) {
    this.narrative = [...narrative];
    this.logs = [];
  }

  /** 执行更新指令 */
  apply(update: NarrativeUpdate): void {
    switch (update.type) {
      case 'add':
        this.narrative.push(update.line);
        this.log(`添加叙事行: ${update.line.type}${update.line.metadata?.actionType ? `(${update.line.metadata.actionType})` : ''}`);
        break;
        
      case 'expireActionType':
        const expiredCount = this.expireByActionType(update.actionType);
        this.log(`过期 ${update.actionType} 类型卡片: ${expiredCount} 个`);
        break;
        
      case 'expireLocation':
        this.expireByLocation(update.locationId);
        this.log(`过期场景 ${update.locationId} 的链接`);
        break;
        
      case 'expireAllActions':
        const allCount = this.expireAllActions();
        this.log(`过期所有操作卡片: ${allCount} 个`);
        break;
        
      case 'expireSceneLinks':
        const sceneCount = this.expireSceneLinks();
        this.log(`过期场景链接: ${sceneCount} 个`);
        break;
    }
  }

  /** 批量执行 */
  applyAll(updates: NarrativeUpdate[]): void {
    for (const update of updates) {
      this.apply(update);
    }
  }

  /** 获取结果 */
  getResult(): { narrative: NarrativeLine[]; logs: string[] } {
    return {
      narrative: this.narrative,
      logs: this.logs,
    };
  }

  // ============ 私有方法 ============

  private expireByActionType(actionType: string): number {
    let count = 0;
    this.narrative = this.narrative.map(line => {
      if (line.metadata?.actionType === actionType && !line.expired) {
        count++;
        return { ...line, expired: true };
      }
      return line;
    });
    return count;
  }

  private expireByLocation(locationId: string): void {
    this.narrative = this.narrative.map(line => {
      if (line.metadata?.locationId === locationId && !line.expired) {
        return { ...line, expired: true };
      }
      return line;
    });
  }

  private expireAllActions(): number {
    let count = 0;
    this.narrative = this.narrative.map(line => {
      if (line.metadata?.actionType && !line.expired) {
        count++;
        return { ...line, expired: true };
      }
      return line;
    });
    return count;
  }

  private expireSceneLinks(): number {
    let count = 0;
    this.narrative = this.narrative.map(line => {
      if (line.type === 'scene' && line.links && line.links.length > 0 && !line.expired) {
        count++;
        return { ...line, expired: true };
      }
      return line;
    });
    return count;
  }

  private log(message: string): void {
    this.logs.push(`[${new Date().toISOString()}] ${message}`);
  }
}
```

### 4. Reducer 中使用

```typescript
// ============ contexts/GameContext.tsx ============

import { NarrativeManager } from '@/managers/NarrativeManager';
import { OPERATION_CONFIG, PROGRESS_COMPLETE_CONFIG } from '@/config/operations';

case 'BUY_ITEM': {
  if (action.itemId === 'water') {
    // 1. 执行业务逻辑
    if (state.player.money < 2) {
      return addSystemMessage(state, '你的余额不足。');
    }
    
    const newInv = updateInventory(state.inventory, 'water', 1);
    const newPlayer = { ...state.player, money: state.player.money - 2 };
    
    // 2. 使用 NarrativeManager 处理叙事更新
    const manager = new NarrativeManager(state.narrative);
    const config = OPERATION_CONFIG.BUY_ITEM;
    
    // 过期之前的购买卡片
    for (const actionType of config.expireActionTypes) {
      manager.apply({ type: 'expireActionType', actionType });
    }
    
    // 过期原始场景链接
    if (config.expireCurrentScene) {
      manager.apply({ type: 'expireSceneLinks' });
    }
    
    // 添加系统消息
    if (config.systemMessage) {
      manager.apply({
        type: 'add',
        line: createSystemLine(config.systemMessage),
      });
    }
    
    // 添加交互卡片
    if (config.showActionCard) {
      const loc = state.locations[state.currentLocation];
      const card = createActionCard(loc, 'buy');
      if (card) {
        manager.apply({ type: 'add', line: card });
      }
    }
    
    // 3. 获取结果和日志
    const { narrative, logs } = manager.getResult();
    
    // 4. 开发模式下打印日志
    if (process.env.NODE_ENV === 'development') {
      console.group(`[BUY_ITEM]`);
      logs.forEach(log => console.log(log));
      console.groupEnd();
    }
    
    return {
      ...state,
      player: newPlayer,
      inventory: newInv,
      narrative,
    };
  }
  return state;
}

case 'GOTO_LOCATION': {
  const loc = state.locations[action.locationId];
  if (!loc) return state;
  
  // 计算体能消耗
  const isIndoor = loc.type === 'indoor' && state.locations[state.currentLocation]?.type === 'indoor';
  const cost = isIndoor ? 1 : 3;
  const newStamina = Math.max(0, state.player.stamina - cost);
  
  // 使用 NarrativeManager
  const manager = new NarrativeManager(state.narrative);
  const config = OPERATION_CONFIG.GOTO_LOCATION;
  
  // 过期所有操作卡片
  if (config.expireAllActions) {
    manager.apply({ type: 'expireAllActions' });
  }
  
  // 过期场景链接
  if (config.expireCurrentScene) {
    manager.apply({ type: 'expireSceneLinks' });
  }
  
  // 添加新场景
  manager.apply({
    type: 'add',
    line: createSceneLine(loc),
  });
  
  const { narrative, logs } = manager.getResult();
  
  if (process.env.NODE_ENV === 'development') {
    console.group(`[GOTO_LOCATION: ${action.locationId}]`);
    logs.forEach(log => console.log(log));
    console.groupEnd();
  }
  
  return {
    ...state,
    currentLocation: action.locationId,
    player: { ...state.player, stamina: newStamina },
    narrative,
  };
}
```

### 5. 工具函数

```typescript
// ============ utils/narrativeUtils.ts ============

let idCounter = 0;

export function createLineId(): string {
  return `n_${++idCounter}_${Date.now().toString(36)}`;
}

export function createSystemLine(text: string): NarrativeLine {
  return {
    id: createLineId(),
    type: 'system',
    text,
    metadata: { timestamp: Date.now() },
  };
}

export function createSceneLine(loc: LocationNode): NarrativeLine {
  return {
    id: createLineId(),
    type: 'scene',
    text: loc.description,
    links: loc.actions?.map(a => ({ label: a.label, command: a.command })),
    metadata: {
      locationId: loc.id,
      actionType: 'scene',
      timestamp: Date.now(),
    },
  };
}

export function createActionCard(
  loc: LocationNode,
  actionType: string
): NarrativeLine | null {
  const config = loc.onActionComplete?.[actionType as keyof typeof loc.onActionComplete];
  if (!config) return null;
  
  return {
    id: createLineId(),
    type: 'scene',
    text: config.text,
    links: config.links,
    metadata: {
      locationId: loc.id,
      actionType,
      timestamp: Date.now(),
    },
  };
}
```

### 第一步：创建新文件（不改动现有代码）

```
src/
  types/
    narrative.ts          # 新类型定义
  config/
    operations.ts         # 操作配置表
  managers/
    NarrativeManager.ts   # 叙事管理器
```

### 第二步：逐个迁移 reducer

1. 先迁移 `GOTO_LOCATION`（最简单）
2. 再迁移 `BUY_ITEM`
3. 再迁移 `EAT_FOOD` / `COMPLETE_PROGRESS`
4. 最后迁移其他操作
